A. INITIALISATION

Fonction: init_db()
──────────────────
Description: Initialise la base de données avec toutes les tables
Retour: None
Effets: Crée absences.db (SQLite) ou tables MySQL
Erreurs: Ignore si tables existent déjà

Exemple:
    from database import init_db
    init_db()  # Base est prête


Fonction: get_connection()
──────────────────────────
Description: Retourne une connexion à la base de données
Retour: sqlite3.Connection (SQLite) ou mysql.connector.MySQLConnection (MySQL)
Utilisation: Requêtes personnalisées avancées

Exemple:
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM etudiants")
    results = cursor.fetchall()

═════════════════════════════════════════════════════════════════════════════════
B. UTILISATEURS (ADMIN) - ANCIEN SYSTÈME
═════════════════════════════════════════════════════════════════════════════════

Fonction: verify_login(username, password)
───────────────────────────────────────────
Description: Vérifie les identifiants administrateur
Arguments:
  - username: str (nom d'utilisateur)
  - password: str (mot de passe en clair)
Retour: dict (admin user) ou None
Erreurs: None si identifiants incorrects

Exemple:
    admin = verify_login("admin", "password123")
    if admin:
        print(f"Connecté: {admin['username']}")
    else:
        print("Identifiants incorrects")


Fonction: add_admin(username, password)
───────────────────────────────────────
Description: Ajoute un nouvel administrateur
Arguments:
  - username: str (unique)
  - password: str (haché en SHA256)
Retour: bool (succès ou non)
Erreurs: False si username existe déjà

Exemple:
    if add_admin("prof_jean", "motdepasse123"):
        print("Admin créé")


Fonction: update_admin_password(username, new_password)
──────────────────────────────────────────────────────
Description: Change le mot de passe d'un admin
Arguments:
  - username: str
  - new_password: str (haché automatiquement)
Retour: bool
Erreurs: False si utilisateur n'existe pas

Exemple:
    update_admin_password("admin", "nouveau_motdepasse")


Fonction: delete_admin(username)
────────────────────────────────
Description: Supprime un administrateur
Arguments:
  - username: str
Retour: bool
Erreurs: False si utilisateur n'existe pas

Exemple:
    delete_admin("ancien_prof")

Erreurs: Liste vide si erreur

D. ABSENCES
═════════════════════════════════════════════════════════════════════════════════

Fonction: add_absence(student_id, date, is_present)
────────────────────────────────────────────────────
Description: Enregistre une absence/présence
Arguments:
  - student_id: int
  - date: str (format "YYYY-MM-DD")
  - is_present: bool (True = présent, False = absent)
Retour: int (id absence) ou None
Erreurs: None si erreur

Exemple:
    absence_id = add_absence(5, "2024-01-15", False)  # Jean absent le 15 jan
    presence_id = add_absence(5, "2024-01-16", True)  # Présent le 16 jan


Fonction: get_absences_by_student(student_id)
──────────────────────────────────────────────
Description: Récupère toutes les absences d'un étudiant
Arguments:
  - student_id: int
Retour: list[dict] avec colonnes (id, student_id, date, is_present)
Erreurs: Liste vide si aucune

Exemple:
    absences = get_absences_by_student(5)
    for absence in absences:
        status = "Présent" if absence['is_present'] else "Absent"
        print(f"{absence['date']}: {status}")


Fonction: get_absences_by_date(date)
─────────────────────────────────────
Description: Récupère tous les enregistrements pour une date
Arguments:
  - date: str (format "YYYY-MM-DD")
Retour: list[dict]
Erreurs: Liste vide si aucun

Exemple:
    jan15 = get_absences_by_date("2024-01-15")
    print(f"Nombre d'absents le 15: {sum(1 for a in jan15 if not a['is_present'])}")


Fonction: get_absences_by_date_range(start_date, end_date)
───────────────────────────────────────────────────────────
Description: Récupère les absences sur une période
Arguments:
  - start_date: str (format "YYYY-MM-DD")
  - end_date: str (format "YYYY-MM-DD")
Retour: list[dict]
Erreurs: Liste vide si aucun

Exemple:
    jan_absences = get_absences_by_date_range("2024-01-01", "2024-01-31")


Fonction: delete_absence(absence_id)
─────────────────────────────────────
Description: Supprime un enregistrement d'absence
Arguments:
  - absence_id: int
Retour: bool
Erreurs: False si ID invalide

Exemple:
    if delete_absence(42):
        print("Enregistrement supprimé")


Fonction: count_absences_by_student(student_id)
───────────────────────────────────────────────
Description: Compte les absences d'un étudiant
Arguments:
  - student_id: int
Retour: int (nombre d'absences)
Erreurs: 0 si erreur

Exemple:

Fonction: get_presence_rate_by_student(student_id)
──────────────────────────────────────────────────
Description: Calcule le taux de présence

  - student_id: int

Erreurs: 0.0 si erreur ou aucunedonnée

Exemple:

═════════════════════════════════════════════════════════════════════════════════
G. AUTHENTIFICATION 2FA (TOTP)
═════════════════════════════════════════════════════════════════════════════════

Fonction: enable_2fa(admin_id)
──────────────────────────────
Description: Active l'authentification 2FA pour un admin
Arguments:
  - admin_id: int
Retour: dict avec clés:
  - secret: str (secret TOTP à scanner)
  - qr_code: str (URL pour générer QR code)
Erreurs: None si erreur

Exemple:
    result = enable_2fa(3)
    if result:
        print(f"Secret: {result['secret']}")
        print(f"QR Code: {result['qr_code']}")
        print("Scanner le QR code avec Google Authenticator")


Fonction: disable_2fa(admin_id)
───────────────────────────────
Description: Désactive la 2FA pour un admin
Arguments:
  - admin_id: int
Retour: bool
Erreurs: False si ID invalide

Exemple:
    disable_2fa(3)


Fonction: verify_2fa_token(admin_id, token)
────────────────────────────────────────────
Description: Vérifie un code TOTP à 6 chiffres
Arguments:
  - admin_id: int
  - token: str (6 chiffres)
Retour: bool (True = valide)
Erreurs: False si invalide

Exemple:
    if verify_2fa_token(3, "123456"):
        print("Connexion sécurisée")
    else:
        print("Code incorrect")

Fonction: log_action(admin_id, action, table_name, record_id, old_value, new_value)
───────────────────────────────────────────────────────────────────────────────────
Description: Enregistre une action pour audit
Arguments:
  - admin_id: int (qui a fait l'action)
  - action: str ("CREATE" | "UPDATE" | "DELETE" | "VIEW")
  - table_name: str (ex: "etudiants")
  - record_id: int (id de l'enregistrement modifié)
  - old_value: str (ancienne valeur, peut être None)
  - new_value: str (nouvelle valeur, peut être None)
Retour: int (id log) ou None
Erreurs: None si erreur

Exemple:
    log_action(2, "UPDATE", "etudiants", 5, "Jean Dupont", "Jean Durand")


Fonction: get_audit_logs(admin_id=None, action=None, table_name=None)
─────────────────────────────────────────────────────────────────────
Description: Récupère les logs d'audit filtrés
Arguments:
  - admin_id: int ou None (tous si None)
  - action: str ou None (tous si None)
  - table_name: str ou None (tous si None)
Retour: list[dict]
Erreurs: Liste vide si aucun

Exemple:
    # Tous les logs
    all_logs = get_audit_logs()
    
    # Logs de suppression par l'admin 2
    deletes = get_audit_logs(admin_id=2, action="DELETE")

Exemple:
    detail = get_audit_log_detail(150)
    print(f"Avant: {detail['old_value']}")
    print(f"Après: {detail['new_value']}")
